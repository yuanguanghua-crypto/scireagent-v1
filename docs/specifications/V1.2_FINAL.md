# 1. Serializer Field Corrections (from Round 5 Audit)

## 1.1 Required Serializers (Per-Model, No Cross-Model Reuse)

```python
class ApplicationBriefSerializer(serializers.ModelSerializer):
    class Meta:
        model = Application
        fields = ['id', 'name', 'slug', 'summary']  # NOT description

class MethodBriefSerializer(serializers.ModelSerializer):
    class Meta:
        model = Method
        fields = ['id', 'name', 'slug', 'purpose']

class ProtocolBriefSerializer(serializers.ModelSerializer):
    estimated_time_minutes = serializers.SerializerMethodField()

    class Meta:
        model = Protocol
        fields = ['id', 'name', 'slug', 'objective', 'estimated_time_minutes']

    def get_estimated_time_minutes(self, obj):
        from django.db.models import Sum
        from math import ceil
        total = obj.steps.aggregate(total=Sum('duration_seconds'))['total'] or 0
        return ceil(total / 60)

class ProductBriefSerializer(serializers.ModelSerializer):
    catalog_no = serializers.CharField()

    class Meta:
        model = Product
        fields = ['id', 'name', 'slug', 'catalog_no', 'cas']

class RelatedProductSerializer(serializers.Serializer):
    id = serializers.IntegerField()
    name = serializers.CharField()
    catalog_no = serializers.CharField()
    cas = serializers.CharField()
    match_reason = serializers.CharField()

class FAQSerializer(serializers.Serializer):
    question = serializers.CharField()
    answer = serializers.CharField()

class GraphNodeSerializer(serializers.Serializer):
    id = serializers.CharField()
    type = serializers.CharField()
    label = serializers.CharField()
    slug = serializers.CharField(required=False)

class GraphEdgeSerializer(serializers.Serializer):
    id = serializers.CharField()
    source = serializers.CharField()
    target = serializers.CharField()
    relationship = serializers.CharField()
```

## 1.2 Field Name Rules

| Wrong | Correct | Model |
|-------|---------|-------|
| `description` | `summary` | Application |
| `description` | `purpose` | Method |
| `estimated_time_minutes` (model field) | `estimated_time_minutes` (SerializerMethodField) | Protocol |
| `product_count` | `products_count` | All annotations |
| `fields = '__all__'` | Explicit field list | All serializers |

```python
# apps/commerce/services/faq_service.py

def generate_faq(product, applications, methods, protocols):
    """Generate FAQ list from existing data. Returns list of {question, answer}."""
    faq = []

    if applications:
        faq.append({
            'question': f'What is {product.name} used for?',
            'answer': ', '.join(app.name for app in applications)
        })

    if methods:
        faq.append({
            'question': f'Which methods use {product.name}?',
            'answer': ', '.join(m.name for m in methods)
        })

    if product.storage:
        faq.append({
            'question': f'How should {product.name} be stored?',
            'answer': product.storage
        })

    if protocols:
        faq.append({
            'question': f'Which protocols use {product.name}?',
            'answer': ', '.join(p.name for p in protocols)
        })

    return faq
```

## 4.3 Product Relationship Service

```python
# apps/commerce/services/product_relationship_service.py

from django.db.models import Case, When, Value, IntegerField, Q
from apps.bridges.models import ProductMethod, MethodProtocol

def get_related_products(product, limit=4):
    """
    Priority: same_application(+100) > same_method(+50) > same_protocol(+20) > same_product_class(+10)
    """
    # Get this product's method IDs
    method_ids = list(ProductMethod.objects.filter(product=product).values_list('method_id', flat=True))

    # Get this product's application IDs
    app_ids = list(Application.objects.filter(methods__id__in=method_ids).values_list('id', flat=True).distinct())

    # Get protocol IDs via MethodProtocol
    protocol_ids = list(MethodProtocol.objects.filter(method_id__in=method_ids).values_list('protocol_id', flat=True).distinct())

    # Find candidate product IDs
    same_app = set(ProductMethod.objects.filter(
        method__application__id__in=app_ids
    ).exclude(product=product).values_list('product_id', flat=True))

    same_method = set(ProductMethod.objects.filter(
        method_id__in=method_ids
    ).exclude(product=product).values_list('product_id', flat=True))

    same_protocol = set(MethodProtocol.objects.filter(
        protocol_id__in=protocol_ids
    ).values_list('method_id', flat=True))
    same_protocol_products = set(ProductMethod.objects.filter(
        method_id__in=same_protocol
    ).exclude(product=product).values_list('product_id', flat=True))

    same_class = set(Product.objects.filter(
        product_class=product.product_class
    ).exclude(id=product.id).values_list('id', flat=True))

    # Score each candidate
    scores = {}
    for pid in same_app:
        scores[pid] = scores.get(pid, 0) + 100
    for pid in same_method:
        scores[pid] = scores.get(pid, 0) + 50
    for pid in same_protocol_products:
        scores[pid] = scores.get(pid, 0) + 20
    for pid in same_class:
        scores[pid] = scores.get(pid, 0) + 10

    if not scores:
        return Product.objects.none()

    # Sort by score descending, take top N
    sorted_ids = sorted(scores.keys(), key=lambda pid: scores[pid], reverse=True)[:limit]

    # Build queryset with match_reason
    products = Product.objects.filter(id__in=sorted_ids, status='active')
    result = []
    for p in products:
        score = scores[p.id]
        if score >= 100:
            reason = 'Same Application'
        elif score >= 50:
            reason = 'Same Method'
        elif score >= 20:
            reason = 'Same Protocol'
        else:
            reason = 'Same Category'
        result.append({
            'id': p.id,
            'name': p.name,
            'catalog_no': p.catalog_no,
            'cas': p.cas,
            'match_reason': reason,
        })

    # Sort by score
    result.sort(key=lambda x: scores[x['id']], reverse=True)
    return result
```
